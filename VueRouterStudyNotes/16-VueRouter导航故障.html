Vue Router еҜјиҲӘж•…йҡң

жЈҖжөӢеҜјиҲӘж•…йҡң

еҰӮжһңеҜјиҲӘиў«йҳ»жӯўпјҢеҜјиҮҙз”ЁжҲ·еҒңз•ҷеңЁеҗҢдёҖдёӘйЎөйқўдёҠпјҢз”ұ router.push иҝ”еӣһзҡ„ Promise зҡ„и§ЈжһҗеҖје°ҶжҳҜ Navigation FailureгҖӮеҗҰеҲҷпјҢе®ғе°ҶжҳҜдёҖдёӘ falsy еҖј (йҖҡеёёжҳҜ undefined)гҖӮ

const navigationResult = await router.push('/test')

if (navigationResult) {
  // еҜјиҲӘиў«йҳ»жӯў
} else {
  // еҜјиҲӘжҲҗеҠҹ (еҢ…жӢ¬йҮҚж–°еҜјиҲӘзҡ„жғ…еҶө)
}

Navigation Failure жҳҜеёҰжңүдёҖдәӣйўқеӨ–еұһжҖ§зҡ„ Error е®һдҫӢпјҢиҝҷдәӣеұһжҖ§еҸҜд»Ҙз”ЁжқҘеҢәеҲҶдёҚеҗҢзҡ„еҜјиҲӘж•…йҡңгҖӮ

иҰҒжЈҖжҹҘеҜјиҲӘз»“жһңзҡ„жҖ§иҙЁпјҢеҸҜд»ҘдҪҝз”Ё isNavigationFailure еҮҪж•°гҖӮ

import { NavigationFailureType, isNavigationFailure } from 'vue-router'

// иҜ•еӣҫзҰ»ејҖжңӘдҝқеӯҳзҡ„зј–иҫ‘ж–Үжң¬з•Ңйқў
const failure = await router.push('/articles')

if (isNavigationFailure(failure, NavigationFailureType.aborted)) {
  // з»ҷз”ЁжҲ·жҳҫзӨәдёҖдёӘе°ҸйҖҡзҹҘ
  showToast('You have unsaved changes, discard and leave anyway?')
}

е…ЁеұҖеҜјиҲӘж•…йҡң

еҸҜд»ҘдҪҝз”Ё router.afterEach() еҜјиҲӘе®ҲеҚ«жқҘжЈҖжөӢе…ЁеұҖеҜјиҲӘж•…йҡңгҖӮ

router.afterEach((to, from, failure) => {
  if (failure) {
    // еҜјиҲӘиў«йҳ»жӯў
  }else {
    // еҜјиҲӘжҲҗеҠҹ
  }
})

йүҙеҲ«еҜјиҲӘж•…йҡң

жңүдёҚеҗҢзҡ„жғ…еҶөдјҡеҜјиҮҙеҜјиҲӘзҡ„дёӯжӯўпјҢжүҖжңүиҝҷдәӣжғ…еҶөйғҪдјҡеҜјиҮҙдёҚеҗҢзҡ„ Navigation FailureгҖӮе®ғд»¬еҸҜд»Ҙз”Ё isNavigationFailure е’Ң NavigationFailureType жқҘеҢәеҲҶгҖӮжҖ»е…ұжңүдёүз§ҚдёҚеҗҢзҡ„зұ»еһӢпјҡ

  • abortedпјҡеңЁеҜјиҲӘе®ҲеҚ«дёӯиҝ”еӣһ false дёӯж–ӯдәҶжң¬ж¬ЎеҜјиҲӘгҖӮ
  • cancelledпјҡеңЁеҪ“еүҚеҜјиҲӘе®ҢжҲҗд№ӢеүҚеҸҲжңүдәҶдёҖдёӘж–°зҡ„еҜјиҲӘгҖӮжҜ”еҰӮпјҢеңЁзӯүеҫ…еҜјиҲӘе®ҲеҚ«зҡ„иҝҮзЁӢдёӯеҸҲи°ғз”ЁдәҶ router.pushгҖӮ
  • duplicatedпјҡеҜјиҲӘиў«йҳ»жӯўпјҢеӣ дёәжҲ‘д»¬е·Із»ҸеңЁзӣ®ж ҮдҪҚзҪ®дәҶгҖӮ

еҜјиҲӘж•…йҡңзҡ„еұһжҖ§

жүҖжңүзҡ„еҜјиҲӘеӨұиҙҘйғҪдјҡжҡҙйңІ to е’Ң from еұһжҖ§пјҢд»ҘеҸҚжҳ еӨұиҙҘеҜјиҲӘзҡ„еҪ“еүҚдҪҚзҪ®е’Ңзӣ®ж ҮдҪҚзҪ®гҖӮ

// жӯЈеңЁе°қиҜ•и®ҝй—® admin йЎөйқў
router.push('/admin').then(failure => {
  if (isNavigationFailure(failure, NavigationFailureType.aborted)) {
    failure.to.path // '/admin'
    failure.from.path // '/'
  }
})

жЈҖжөӢйҮҚе®ҡеҗ‘

йҖҡиҝҮиҜ»еҸ–и·Ҝз”ұең°еқҖдёӯзҡ„ redirectedFrom еұһжҖ§пјҢеҜ№е…¶иҝӣиЎҢдёҚеҗҢзҡ„жЈҖжҹҘгҖӮ

await router.push('/test')
if (router.currentRoute.value.redirectedFrom) {
  // redirectedFrom жҳҜи§ЈжһҗеҮәзҡ„и·Ҝз”ұең°еқҖпјҢе°ұеғҸеҜјиҲӘе®ҲеҚ«дёӯзҡ„ toе’Ң from
}
