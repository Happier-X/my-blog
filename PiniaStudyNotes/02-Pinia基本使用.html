Pinia еҹәжң¬дҪҝз”Ё

Happier2024е№ҙ10жңҲ30ж—Ҙ...еӨ§зәҰ 1 еҲҶй’ҹејҖеҸ‘VuePinia

е®ҡд№ү Store

Store жҳҜз”Ё defineStore() е®ҡд№үзҡ„пјҢе®ғзҡ„з¬¬дёҖдёӘеҸӮж•°иҰҒжұӮжҳҜдёҖдёӘзӢ¬дёҖж— дәҢзҡ„еӯ—з¬ҰдёІпјҢдҪңдёә Store зҡ„ idпјҢз¬¬дәҢдёӘеҸӮж•°жҳҜ Setup еҮҪж•° жҲ– Option еҜ№иұЎ (иҝҷйҮҢд»Ҙ Setup еҮҪж•° дёәдҫӢ)пјҢз”ЁдәҺй…ҚзҪ® StoreгҖӮ

// src/stores/counter.js
import { defineStore } from 'pinia'
import { ref, computed, inject } from 'vue'
import { useRoute, useRouter } from 'vue-router'

// еҸҜд»Ҙд»»ж„Ҹе‘ҪеҗҚ defineStore() зҡ„иҝ”еӣһеҖјпјҢдҪҶжңҖеҘҪдҪҝз”Ё use еүҚзјҖпјҢдҪҝз”Ё Store з»“е°ҫ
// з¬¬дёҖдёӘеҸӮж•°жҳҜ Store зҡ„ idпјҢе®ғиҰҒжұӮжҳҜе”ҜдёҖзҡ„пјҢз¬¬дәҢдёӘеҸӮж•°жҳҜ Setup еҮҪж•°
export const useCounterStore = defineStore('counter', () => {
    // е®ҡд№үзҠ¶жҖҒпјҲStateпјү
    const count = ref(0)
    // е®ҡд№үи®Ўз®—еұһжҖ§пјҲGettersпјү
    const doubleCount = computed(() => count.value * 2)
    // е®ҡд№үж–№жі•пјҲActionsпјү
    const increment = () => {
        count.value++
    }

    // д№ҹеҸҜд»Ҙдҫқиө–дәҺе…ЁеұҖжҸҗдҫӣзҡ„еұһжҖ§
    const route = useRoute()
    const router = useRouter()
    const appProvided = inject('appProvided')

    // иҝ”еӣһзҠ¶жҖҒгҖҒи®Ўз®—еұһжҖ§е’Ңж–№жі•
    return { count, doubleCount, increment }
})

дҪҝз”Ё Store

<script setup>
// еҜје…Ҙе®ҡд№үзҡ„ Store
import { useCounterStore } from "@/stores/counter";
// иҺ·еҸ– Store
const counterStore = useCounterStore();
// д»Һ Store дёӯиҺ·еҸ–зҠ¶жҖҒгҖҒи®Ўз®—еұһжҖ§е’Ңж–№жі•
const count = counterStore.count;
const doubleCount = counterStore.doubleCount;
const increment = counterStore.increment;
</script>

еҰӮжһңзӣҙжҺҘи§Јжһ„ Store дёӯзҡ„зҠ¶жҖҒе’Ңи®Ўз®—еұһжҖ§пјҢдјҡдёўеӨұе“Қеә”ејҸпјҢжүҖд»ҘйңҖиҰҒдҪҝз”Ё storeToRefs() еҮҪж•°гҖӮ

<script setup>
import { storeToRefs } from "pinia";
import { useCounterStore } from "@/stores/counter";
const counterStore = useCounterStore();
// дҪҝз”Ё `storeToRefs()` еҮҪж•°и§Јжһ„зҠ¶жҖҒе’Ңи®Ўз®—еұһжҖ§
const { count, doubleCount } = storeToRefs(counterStore);
// ж–№жі•еҸҜд»ҘзӣҙжҺҘи§Јжһ„
const { increment } = counterStore;
</script>

е»әи®®дёҚдҪҝз”Ёи§Јжһ„иөӢеҖјзҡ„ж–№ејҸпјҢеёҰдёҠ Store д»Јз ҒеҸҜиҜ»жҖ§д№ҹдјҡжӣҙеҘҪдёҖзӮ№гҖӮ
