Vue Router е‘ҪеҗҚи·Ҝз”ұ

Happier2024е№ҙ10жңҲ25ж—Ҙ...е°ҸдәҺ 1 еҲҶй’ҹејҖеҸ‘VueVue Router

йҖҡиҝҮз»ҷи·Ҝз”ұе‘ҪеҗҚпјҢеҸҜд»Ҙжӣҙж–№дҫҝзҡ„и·іиҪ¬и·Ҝз”ұгҖӮ

дҪҝз”Ё name еұһжҖ§з»ҷи·Ҝз”ұе‘ҪеҗҚпјҢе‘ҪеҗҚзҡ„и·Ҝз”ұеҗҚз§°еҸҜд»ҘиҮӘе®ҡд№үдҪҶжҳҜиҰҒе”ҜдёҖгҖӮ

йҖҡиҝҮ :to="{ name: 'и·Ҝз”ұеҗҚз§°' }" и·іиҪ¬и·Ҝз”ұгҖӮ

д»Јз ҒзӨәдҫӢеҰӮдёӢгҖӮ

// src/router/index.js
import { createWebHistory, createRouter } from 'vue-router'

import HomeView from '../views/HomeView.vue'
import AboutView from '../views/AboutView.vue'

const routes = [
    {
        path: '/',
        component: HomeView
    },
    {
        path: '/about',
        // йҖҡиҝҮ name з»ҷи·Ҝз”ұе‘ҪеҗҚ
        name: 'about',
        component: AboutView
    }
]

const router = createRouter({
    history: createWebHistory(),
    routes
})

export default router
<!-- src/App.vue -->
<template>
  <h1>жҲ‘жҳҜ APP</h1>
  <nav>
    <RouterLink to="/">Home</RouterLink>
    <span>|</span>
    <RouterLink :to="{ name: 'about' }">About</RouterLink>
  </nav>
  <main>
    <RouterView />
  </main>
</template>
